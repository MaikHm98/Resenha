📘 PROMPT MESTRE — PROJETO RESENHA
Estou desenvolvendo um aplicativo chamado RESENHA.

O objetivo do projeto é criar um sistema para organização de grupos de futebol amador (rachão), com controle de presença, sistema dinâmico de capitães por desafio, votação de MVP e Bola Murcha, classificação por temporada e histórico geral acumulado.

O projeto tem como foco principal aprendizado profundo de arquitetura e boas práticas, utilizando:

- Front-end: React Native (Expo)
- Back-end: ASP.NET Core Web API (.NET 8)
- Banco de Dados: MySQL
- Arquitetura: Controller → Service → Repository
- Banco com nomenclatura em português
- Código com nomenclatura em inglês simples e profissional

Regras principais do sistema:

1. Usuário pode participar de múltiplos grupos.
2. Grupo pode ter múltiplos administradores.
3. Apenas administradores podem criar grupos.
4. Jogadores entram via convite.
5. Sempre existirão 2 capitães por jogo.
6. Sistema controla automaticamente quem pode ser desafiado.
7. Capitão derrotado sai do ciclo.
8. Quando capitão perde, ciclo reinicia.
9. Limite de jogadores bloqueia novas confirmações.
10. Pontuação:
   - Vitória = 3
   - Presença = 1
   - Derrota = 0
11. Classificação por temporada.
12. Histórico geral acumulado.
13. Nova temporada precisa ser confirmada por administrador.
14. Todos votam MVP e Bola Murcha.
15. Não pode votar em si mesmo.
16. Empate abre nova votação.
17. Administradores aprovam resultado final.

O código deve ser limpo, organizado, comentado quando necessário, e fácil de manutenção.
Esse é o DNA do projeto.

Toda decisão futura deve respeitar isso.

-- =========================================================
-- BANCO: RESENHA (modelagem em português do Brasil)
-- Charset recomendado
-- =========================================================

CREATE DATABASE IF NOT EXISTS resenha
  CHARACTER SET utf8mb4
  COLLATE utf8mb4_0900_ai_ci;

USE resenha;

-- =========================================================
-- 1) USUÁRIOS
-- =========================================================
CREATE TABLE IF NOT EXISTS usuarios (
  id_usuario           BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  nome                 VARCHAR(120) NOT NULL,
  email                VARCHAR(180) NOT NULL,
  senha_hash           VARCHAR(255) NOT NULL,
  ativo                TINYINT(1) NOT NULL DEFAULT 1,
  criado_em            DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  atualizado_em        DATETIME NULL DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (id_usuario),
  UNIQUE KEY uq_usuarios_email (email)
) ENGINE=InnoDB;

-- =========================================================
-- 2) GRUPOS
-- Apenas admin cria grupo (regra no back-end)
-- =========================================================
CREATE TABLE IF NOT EXISTS grupos (
  id_grupo             BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  nome                 VARCHAR(120) NOT NULL,
  descricao            VARCHAR(255) NULL,
  limite_jogadores     INT NOT NULL DEFAULT 16,
  criado_por_usuario   BIGINT UNSIGNED NOT NULL,
  ativo                TINYINT(1) NOT NULL DEFAULT 1,
  criado_em            DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  atualizado_em        DATETIME NULL DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (id_grupo),
  KEY ix_grupos_criado_por (criado_por_usuario),
  CONSTRAINT fk_grupos_criado_por
    FOREIGN KEY (criado_por_usuario) REFERENCES usuarios(id_usuario)
) ENGINE=InnoDB;

-- =========================================================
-- 3) USUÁRIOS NO GRUPO (multi-admin)
-- =========================================================
CREATE TABLE IF NOT EXISTS grupo_usuarios (
  id_grupo_usuario     BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  id_grupo             BIGINT UNSIGNED NOT NULL,
  id_usuario           BIGINT UNSIGNED NOT NULL,
  perfil               ENUM('ADMIN','JOGADOR') NOT NULL DEFAULT 'JOGADOR',
  ativo                TINYINT(1) NOT NULL DEFAULT 1,
  entrou_em            DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (id_grupo_usuario),
  UNIQUE KEY uq_grupo_usuarios (id_grupo, id_usuario),
  KEY ix_grupo_usuarios_usuario (id_usuario),
  KEY ix_grupo_usuarios_grupo (id_grupo),
  CONSTRAINT fk_grupo_usuarios_grupo
    FOREIGN KEY (id_grupo) REFERENCES grupos(id_grupo),
  CONSTRAINT fk_grupo_usuarios_usuario
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id_usuario)
) ENGINE=InnoDB;

-- =========================================================
-- 4) CONVITES PARA ENTRAR NO GRUPO (jogador entra via convite)
-- =========================================================
CREATE TABLE IF NOT EXISTS convites_grupo (
  id_convite           BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  id_grupo             BIGINT UNSIGNED NOT NULL,
  email_convidado      VARCHAR(180) NOT NULL,
  codigo_convite       VARCHAR(64) NOT NULL,
  status               ENUM('PENDENTE','ACEITO','EXPIRADO','CANCELADO') NOT NULL DEFAULT 'PENDENTE',
  criado_por_usuario   BIGINT UNSIGNED NOT NULL,
  expira_em            DATETIME NOT NULL,
  criado_em            DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (id_convite),
  UNIQUE KEY uq_convites_codigo (codigo_convite),
  KEY ix_convites_grupo (id_grupo),
  KEY ix_convites_email (email_convidado),
  CONSTRAINT fk_convites_grupo
    FOREIGN KEY (id_grupo) REFERENCES grupos(id_grupo),
  CONSTRAINT fk_convites_criado_por
    FOREIGN KEY (criado_por_usuario) REFERENCES usuarios(id_usuario)
) ENGINE=InnoDB;

-- =========================================================
-- 5) TEMPORADAS (por grupo)
-- Admin confirma criação da temporada
-- =========================================================
CREATE TABLE IF NOT EXISTS temporadas (
  id_temporada         BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  id_grupo             BIGINT UNSIGNED NOT NULL,
  ano                  INT NOT NULL,
  nome                 VARCHAR(60) NOT NULL, -- ex: "Temporada 2026"
  status               ENUM('ATIVA','ENCERRADA','SUGERIDA') NOT NULL DEFAULT 'SUGERIDA',
  confirmada_por_usuario BIGINT UNSIGNED NULL,
  confirmada_em        DATETIME NULL,
  criada_em            DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (id_temporada),
  UNIQUE KEY uq_temporada_grupo_ano (id_grupo, ano),
  KEY ix_temporadas_grupo (id_grupo),
  CONSTRAINT fk_temporadas_grupo
    FOREIGN KEY (id_grupo) REFERENCES grupos(id_grupo),
  CONSTRAINT fk_temporadas_confirmada_por
    FOREIGN KEY (confirmada_por_usuario) REFERENCES usuarios(id_usuario)
) ENGINE=InnoDB;

-- =========================================================
-- 6) PARTIDAS (por grupo e temporada)
-- =========================================================
CREATE TABLE IF NOT EXISTS partidas (
  id_partida           BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  id_grupo             BIGINT UNSIGNED NOT NULL,
  id_temporada         BIGINT UNSIGNED NOT NULL,
  data_hora_jogo       DATETIME NOT NULL,
  status               ENUM('ABERTA','EM_ANDAMENTO','FINALIZADA','CANCELADA') NOT NULL DEFAULT 'ABERTA',
  observacao           VARCHAR(255) NULL,
  criado_por_usuario   BIGINT UNSIGNED NOT NULL,
  criado_em            DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  atualizado_em        DATETIME NULL DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (id_partida),
  KEY ix_partidas_grupo (id_grupo),
  KEY ix_partidas_temporada (id_temporada),
  CONSTRAINT fk_partidas_grupo
    FOREIGN KEY (id_grupo) REFERENCES grupos(id_grupo),
  CONSTRAINT fk_partidas_temporada
    FOREIGN KEY (id_temporada) REFERENCES temporadas(id_temporada),
  CONSTRAINT fk_partidas_criado_por
    FOREIGN KEY (criado_por_usuario) REFERENCES usuarios(id_usuario)
) ENGINE=InnoDB;

-- =========================================================
-- 7) PRESENÇA NA PARTIDA (bloqueia ao atingir limite via regra no Service)
-- =========================================================
CREATE TABLE IF NOT EXISTS presencas_partida (
  id_presenca          BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  id_partida           BIGINT UNSIGNED NOT NULL,
  id_usuario           BIGINT UNSIGNED NOT NULL,
  status               ENUM('CONFIRMADO','CANCELADO') NOT NULL DEFAULT 'CONFIRMADO',
  confirmado_em        DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (id_presenca),
  UNIQUE KEY uq_presenca_partida_usuario (id_partida, id_usuario),
  KEY ix_presencas_usuario (id_usuario),
  CONSTRAINT fk_presencas_partida
    FOREIGN KEY (id_partida) REFERENCES partidas(id_partida),
  CONSTRAINT fk_presencas_usuario
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id_usuario)
) ENGINE=InnoDB;

-- =========================================================
-- 8) SISTEMA DE CAPITÃO (CICLO DE REINADO)
-- Controla quem pode ser desafiado (bloqueados no ciclo atual)
-- =========================================================
CREATE TABLE IF NOT EXISTS ciclos_capitao (
  id_ciclo             BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  id_grupo             BIGINT UNSIGNED NOT NULL,
  id_temporada         BIGINT UNSIGNED NOT NULL,
  id_capitao_atual     BIGINT UNSIGNED NOT NULL,
  status               ENUM('ATIVO','ENCERRADO') NOT NULL DEFAULT 'ATIVO',
  iniciado_em          DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  encerrado_em         DATETIME NULL,
  PRIMARY KEY (id_ciclo),
  KEY ix_ciclos_grupo (id_grupo),
  KEY ix_ciclos_temporada (id_temporada),

Criar Helpers/PasswordHelper.cs

using System.Security.Cryptography;
using System.Text;

namespace Resenha.API.Helpers
{
    public static class PasswordHelper
    {
        public static string HashPassword(string password)
        {
            using var sha256 = SHA256.Create();
            var bytes = Encoding.UTF8.GetBytes(password);
            var hash = sha256.ComputeHash(bytes);
            return Convert.ToBase64String(hash);
        }

        public static bool VerifyPassword(string password, string storedHash)
        {
            var hash = HashPassword(password);
            return hash == storedHash;
        }
    }
}

🧠 4️⃣ AuthService
Services/AuthService.cs

using Microsoft.IdentityModel.Tokens;
using System.IdentityModel.Tokens.Jwt;
using System.Security.Claims;
using System.Text;
using Resenha.API.Data;
using Resenha.API.DTOs;
using Resenha.API.Entities;
using Resenha.API.Helpers;

namespace Resenha.API.Services
{
    public class AuthService
    {
        private readonly ResenhaDbContext _context;
        private readonly IConfiguration _configuration;

        public AuthService(ResenhaDbContext context, IConfiguration configuration)
        {
            _context = context;
            _configuration = configuration;
        }

        public AuthResponseDTO Register(RegisterRequestDTO request)
        {
            if (_context.Usuarios.Any(u => u.Email == request.Email))
                throw new Exception("Email já cadastrado.");

            var usuario = new Usuario
            {
                Nome = request.Nome,
                Email = request.Email,
                SenhaHash = PasswordHelper.HashPassword(request.Senha)
            };

            _context.Usuarios.Add(usuario);
            _context.SaveChanges();

            var token = GenerateJwtToken(usuario);

            return new AuthResponseDTO
            {
                Token = token,
                Nome = usuario.Nome,
                Email = usuario.Email
            };
        }

        public AuthResponseDTO Login(LoginRequestDTO request)
        {
            var usuario = _context.Usuarios
                .FirstOrDefault(u => u.Email == request.Email);

            if (usuario == null)
                throw new Exception("Usuário não encontrado.");

            if (!PasswordHelper.VerifyPassword(request.Senha, usuario.SenhaHash))
                throw new Exception("Senha inválida.");

            var token = GenerateJwtToken(usuario);

            return new AuthResponseDTO
            {
                Token = token,
                Nome = usuario.Nome,
                Email = usuario.Email
            };
        }

        private string GenerateJwtToken(Usuario usuario)
        {
            var jwtSettings = _configuration.GetSection("JwtSettings");

            var key = new SymmetricSecurityKey(
                Encoding.UTF8.GetBytes(jwtSettings["SecretKey"])
            );

            var credentials = new SigningCredentials(key, SecurityAlgorithms.HmacSha256);

            var claims = new[]
            {
                new Claim(ClaimTypes.NameIdentifier, usuario.IdUsuario.ToString()),
                new Claim(ClaimTypes.Email, usuario.Email),
                new Claim(ClaimTypes.Name, usuario.Nome)
            };

            var token = new JwtSecurityToken(
                issuer: jwtSettings["Issuer"],
                audience: jwtSettings["Audience"],
                claims: claims,
                expires: DateTime.UtcNow.AddMinutes(
                    int.Parse(jwtSettings["ExpirationMinutes"])
                ),
                signingCredentials: credentials
            );

            return new JwtSecurityTokenHandler().WriteToken(token);
        }
    }
}

            {

                IdUsuario = userId,
                Perfil = "ADMIN"

            _context.SaveChanges();

public string AcceptInvite(AcceptInviteDTO request, ClaimsPrincipal userClaims)
{
    var userId = ulong.Parse(
        userClaims.FindFirst(ClaimTypes.NameIdentifier)?.Value
    );

    var convite = _context.ConvitesGrupo
        .FirstOrDefault(c =>
            c.CodigoConvite == request.CodigoConvite &&
            c.Status == "PENDENTE");

    if (convite == null)
        throw new Exception("Convite inválido.");

    if (convite.ExpiraEm < DateTime.UtcNow)
        throw new Exception("Convite expirado.");

    var jaExiste = _context.GrupoUsuarios.Any(g =>
        g.IdGrupo == convite.IdGrupo &&
        g.IdUsuario == userId);

    if (jaExiste)
        throw new Exception("Usuário já faz parte do grupo.");

    var grupoUsuario = new GrupoUsuario
    {
        IdGrupo = convite.IdGrupo,
        IdUsuario = userId,
        Perfil = "JOGADOR"
    };

    _context.GrupoUsuarios.Add(grupoUsuario);

    convite.Status = "ACEITO";

    _context.SaveChanges();

    return "Convite aceito com sucesso.";
}

public string ConfirmPresence(ulong idPartida, ClaimsPrincipal userClaims)
{
    var userId = ulong.Parse(
        userClaims.FindFirst(ClaimTypes.NameIdentifier)?.Value
    );

    var partida = _context.Partidas.FirstOrDefault(p => p.IdPartida == idPartida);

    if (partida == null)
        throw new Exception("Partida não encontrada.");

    var pertenceGrupo = _context.GrupoUsuarios.Any(g =>
        g.IdGrupo == partida.IdGrupo &&
        g.IdUsuario == userId &&
        g.Ativo
    );

    if (!pertenceGrupo)
        throw new Exception("Usuário não pertence ao grupo.");

    var jaConfirmado = _context.PresencasPartida.Any(p =>
        p.IdPartida == idPartida &&
        p.IdUsuario == userId
    );

    if (jaConfirmado)
        throw new Exception("Usuário já confirmou presença.");

    var limiteGrupo = _context.Grupos
        .Where(g => g.IdGrupo == partida.IdGrupo)
        .Select(g => g.LimiteJogadores)
        .FirstOrDefault();

    var totalConfirmados = _context.PresencasPartida
        .Count(p => p.IdPartida == idPartida);

    if (totalConfirmados >= limiteGrupo)
        throw new Exception("Limite de jogadores atingido.");

    var presenca = new PresencaPartida
    {
        IdPartida = idPartida,
        IdUsuario = userId
    };

    _context.PresencasPartida.Add(presenca);
    _context.SaveChanges();

    return "Presença confirmada com sucesso.";
}
}

public List<ulong> GetPossibleChallengers(ulong idCiclo)
{
    var ciclo = _context.CiclosCapitao
        .FirstOrDefault(c => c.IdCiclo == idCiclo && c.Status == "ATIVO");

    if (ciclo == null)
        throw new Exception("Ciclo não encontrado.");

    var bloqueados = _context.CiclosCapitaoBloqueados
        .Where(b => b.IdCiclo == idCiclo)
        .Select(b => b.IdUsuarioBloqueado)
        .ToList();

    var jogadoresGrupo = _context.GrupoUsuarios
        .Where(g => g.IdGrupo == ciclo.IdGrupo && g.Ativo)
        .Select(g => g.IdUsuario)
        .ToList();

    var possiveis = jogadoresGrupo
        .Where(u => u != ciclo.IdCapitaoAtual && !bloqueados.Contains(u))
        .ToList();

    return possiveis;
}

        .FirstOrDefault(c => c.IdCiclo == idCiclo && c.Status == "ATIVO");

    if (ciclo == null)
        throw new Exception("Ciclo inválido.");

    if (idCapitaoVencedor == ciclo.IdCapitaoAtual)
    {
        // Capitão venceu → bloquear perdedor
        var bloqueio = new CicloCapitaoBloqueado
        {
            IdCiclo = idCiclo,
            IdUsuarioBloqueado = idCapitaoPerdedor
        };

        _context.CiclosCapitaoBloqueados.Add(bloqueio);
        _context.SaveChanges();

        return "Capitão manteve o reinado.";
    }
    else
    {
        // Capitão perdeu → encerra ciclo atual
        ciclo.Status = "ENCERRADO";
        ciclo.EncerradoEm = DateTime.UtcNow;

        _context.SaveChanges();

        // Criar novo ciclo
        var novoCiclo = new CicloCapitao
        {
            IdGrupo = ciclo.IdGrupo,
            IdTemporada = ciclo.IdTemporada,
            IdCapitaoAtual = idCapitaoVencedor
        };

        _context.CiclosCapitao.Add(novoCiclo);
        _context.SaveChanges();

        return "Novo ciclo iniciado com novo capitão.";
    }
}

Apurar votação
public string CloseAndCalculate(ulong idVotacao)
{
    var votacao = _context.VotacoesPartida
        .FirstOrDefault(v => v.IdVotacao == idVotacao);

    if (votacao == null)
        throw new Exception("Votação não encontrada.");

    var resultado = _context.Votos
        .Where(v => v.IdVotacao == idVotacao)
        .GroupBy(v => v.IdUsuarioVotado)
        .Select(g => new
        {
            Usuario = g.Key,
            Total = g.Count()
        })
        .OrderByDescending(x => x.Total)
        .ToList();

    if (!resultado.Any())
        throw new Exception("Nenhum voto registrado.");

    var maior = resultado.First();
    var empatados = resultado.Where(x => x.Total == maior.Total).ToList();

    if (empatados.Count > 1)
    {
        // Criar nova rodada
        votacao.Status = "APURADA";
        _context.SaveChanges();

        var novaRodada = new VotacaoPartida
        {
            IdPartida = votacao.IdPartida,
            Tipo = votacao.Tipo,
            Rodada = votacao.Rodada + 1
        };

        _context.VotacoesPartida.Add(novaRodada);
        _context.SaveChanges();

        return "Empate detectado. Nova rodada criada.";
    }

    votacao.IdUsuarioVencedorProvisorio = maior.Usuario;
    votacao.Status = "APURADA";

    _context.SaveChanges();

    return "Votação apurada com sucesso.";
}

🔐 4️⃣ Criar AuthContext
src/contexts/AuthContext.js

import React, { createContext, useState } from "react";
import AsyncStorage from "@react-native-async-storage/async-storage";
import api from "../services/api";

export const AuthContext = createContext({});

export function AuthProvider({ children }) {
  const [user, setUser] = useState(null);

  async function login(email, senha) {
    const response = await api.post("/users/login", {
      email,
      senha
    });

    const { token, nome } = response.data;

    await AsyncStorage.setItem("token", token);

    setUser({ nome });
  }

  async function register(nome, email, senha) {
    await api.post("/users/register", {
      nome,
      email,
      senha
    });
  }

  async function logout() {
    await AsyncStorage.removeItem("token");
    setUser(null);
  }

  return (
    <AuthContext.Provider value={{ user, login, register, logout }}>
      {children}
    </AuthContext.Provider>
  );
}

import AsyncStorage from "@react-native-async-storage/async-storage";

});

    </View>
  );
}

import React, { useState } from "react";
import {
  View,
  Text,
  TextInput,
  Button,
  StyleSheet,
  Alert
} from "react-native";
import api from "../services/api";

export default function CreateGroupScreen({ navigation }) {
  const [nome, setNome] = useState("");
  const [descricao, setDescricao] = useState("");
  const [limite, setLimite] = useState("16");
  const [loading, setLoading] = useState(false);

  async function handleCreateGroup() {
    if (!nome) {
      Alert.alert("Erro", "Informe o nome do grupo.");
      return;
    }

    try {
      setLoading(true);

      await api.post("/groups", {
        nome,
        descricao,
        limiteJogadores: parseInt(limite)
      });

      Alert.alert("Sucesso", "Grupo criado com sucesso!");
      navigation.goBack();
    } catch (error) {
      Alert.alert("Erro", "Não foi possível criar o grupo.");
    } finally {
      setLoading(false);
    }
  }

  return (
    <View style={styles.container}>
      <Text style={styles.title}>Criar Novo Grupo</Text>

      <TextInput
        placeholder="Nome do grupo"
        value={nome}
        onChangeText={setNome}
        style={styles.input}
      />

      <TextInput
        placeholder="Descrição"
        value={descricao}
        onChangeText={setDescricao}
        style={styles.input}
      />

      <TextInput
        placeholder="Limite de jogadores"
        value={limite}
        onChangeText={setLimite}
        keyboardType="numeric"
        style={styles.input}
      />

      <Button
        title={loading ? "Criando..." : "Criar Grupo"}
        onPress={handleCreateGroup}
      />
    </View>
  );
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
    padding: 20
  },
  title: {
    fontSize: 22,
    fontWeight: "bold",
    marginBottom: 20
  },
  input: {
    borderWidth: 1,
    borderColor: "#ccc",
    padding: 10,
    marginBottom: 15,
    borderRadius: 6
  }
});
